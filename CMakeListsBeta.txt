cmake_minimum_required(VERSION 3.10)
project(ASCII_Player LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

set(SOURCES
    main.cpp
    ascii_render.cpp
)

set(HEADERS
    ascii_render.hpp
)

# icon resource on Windows only
if (WIN32)
    list(APPEND SOURCES icon.rc)
endif()

find_package(OpenCV REQUIRED)
find_package(Threads REQUIRED)

add_executable(ASCII_Player ${SOURCES} ${HEADERS})

target_include_directories(ASCII_Player PRIVATE
    ${OpenCV_INCLUDE_DIRS}
)

# Link OpenCV and platform libs
if (WIN32)
    target_link_libraries(ASCII_Player PRIVATE ${OpenCV_LIBS})
    # keep the DLL copy step if you ship libvlc.dll with exe
    add_custom_command(TARGET ASCII_Player POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_if_different
            "${CMAKE_SOURCE_DIR}/libvlc.dll"
            "${CMAKE_SOURCE_DIR}/libvlccore.dll"
            $<TARGET_FILE_DIR:ASCII_Player>
    )
    add_custom_command(TARGET ASCII_Player POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_directory
            "${CMAKE_SOURCE_DIR}/plugins"
            "$<TARGET_FILE_DIR:ASCII_Player>/plugins"
    )
else()
    # POSIX: need dl and ncurses (for input), and pthreads via Threads::Threads
    target_link_libraries(ASCII_Player PRIVATE ${OpenCV_LIBS} dl ncurses Threads::Threads)
endif()
